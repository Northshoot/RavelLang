interface Timer():
    implementation:
        c = "rlib/nrf52/app_timer.stg"
    def start_one_shot(time : int32)
    def cancel()
    def start_periodic(time: int32)

    event fired()

interface Led(led_num: int32):
    configuration:
        # 0 - 3
        led_id = led_num
    implementation:
        c = "rlib/nrf52/led.stg"

    def init()
    def on()
    def off()
    def toggle()

controller EController(timer : Timer, timer1 : Timer, led_0 : Led, led_1 : Led):

    event system.started():
        timer.start_periodic(2000)
        timer1.start_periodic(500)



    event timer1.fired():
        led_1.toggle()

    event timer.fired():
        led_0.toggle()


space EmbeddedSpace:
    platform:
        language = CLang
        system = Nrf52.v1

    interfaces:
        timer = Timer()
        timer1 = Timer()
        led0 = Led(led_num = 0)
        led1 = Led(led_num = 1)

    controllers:
        rndcnt_1 = EController(timer = interfaces.timer, timer1 = interfaces.timer, led_1 = interfaces.led1, led_0 = interfaces.led0)
